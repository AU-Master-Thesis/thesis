	\def\halfTileSize{0.5 * \tileSize} % 5.0
	\def\halfPathWidth{0.5 * \pathWidth} % 2.5

	\pgfmathsetlengthmacro{\untilPath}{\halfTileSize - \halfPathWidth} % 5.0 - 2.5 = 2.5

	\draw[rrt line, environment] (0, 0)
						-- ++(\tileSize, 0)
						-- ++(\tileSize, 0)
						-- ++(\untilPath, 0)

						-- ++(-\withPath, 0)
						-- ++(0, -\withPath)
						-- ++(0, -\untilPath)
						-- ++(-\untilPath, 0)
						-- cycle;

	\draw[rrt line, environment] (2 * \tileSize + \untilPath + \pathWidth, 0)
							-- ++(\untilPath, 0)
							-- ++(5 * \tileSize, 0)
							-- ++(0, -\untilPath)
							-- ++(-\withPath, 0)
							-- ++(0, -\withPath)
							-- ++(0, -\untilPath)
							-- ++(-\untilPath, 0)
							-- ++(-\untilPath, 0)
							-- ++(0, \untilPath)
							-- ++(0, \withPath)
							-- ++(-\withPath, 0)
							-- ++(-3 * \tileSize, 0)
							-- ++(-\untilPath, 0)
							-- cycle;

	% ┌─┼─┬─┐┌
	% ┼─┘┌┼┬┼┘
	% ┴┬─┴┼┘│
	% ┌┴┐┌┼─┴┬
	% ├─┴┘└──┘

	\draw[rrt line, environment] (\withPath, -\withPath)
		rectangle ++(2 * \untilPath + \tileSize, -2 * \untilPath);

	\draw[rrt line, environment] (4 * \tileSize + \withPath, -\withPath)
		rectangle ++(2 * \untilPath + \tileSize, -2 * \untilPath);

	\draw[rrt line, environment] (2 * \tileSize + \withPath, -\withPath)
		                    -- ++(2 * \untilPath + \tileSize, 0)
							-- ++(0, -2 * \untilPath)
							-- ++(-\untilPath -\withPath, 0)
							-- ++(0, -\withPath -\untilPath)
							-- ++(-2 * \untilPath - 2* \tileSize, 0)
							-- ++(0, 2 * \untilPath)
							-- ++(\untilPath + \withPath + \tileSize, 0)
							-- ++(0, \withPath + \untilPath);

		                    -- ++(\untilPath, 0)
							-- ++(0, -2 * \untilPath - 2 * \tileSize)
							-- ++(-\tileSize -\untilPath, 0)
							-- ++(0, 2 * \untilPath + \tileSize)
							-- ++(\untilPath + \withPath, 0)

							-- cycle;

	% ├─┴┘└──┘

		                    -- ++(2 * \untilPath + 2 * \tileSize, 0)
